library(ggplot2)
library(tidyverse)
library(shiny)
library(bslib)
scottish_exports <- CodeClanData::scottish_exports 

scottish_exports%>% 
  ggplot(aes(x = year,
      y = exports,
      colour = sector))+
  geom_line()+
  geom_text(data = scottish_exports %>% 
              filter(year == "2017"),
            aes(label = sector),
            check_overlap = TRUE)+
  theme_linedraw()+
  labs(x = "Export volume",
       y = "Year")


?geom_text
scottish_exports %>% 
  group_by(year) %>% 
  mutate(proportion_exports = exports/sum(exports)) %>% 
  ggplot(aes(x = year,
             y = proportion_exports,
             fill = sector))+
  geom_col(position = "stack")+
  theme_linedraw()
